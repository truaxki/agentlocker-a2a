## TL;DR

The A2A SDK exists in two official flavors:

| | Python | TypeScript |
|---|---|---|
| **Package** | `a2a-sdk` | `@a2a-js/sdk` |
| **Version** | 1.0.3 (May 2026) | 0.3.13 (Mar 2026) |
| **Protocol** | A2A v1.0.0 | A2A v0.3.0 |
| **Types** | Protobuf (v1.0) | TypeScript discriminated unions |
| **Server** | Starlette / Uvicorn | Express middleware |
| **Client** | `create_client()` factory | `ClientFactory.createFromUrl()` |
| **Streaming** | SSE via `httpx-sse` | SSE via `ReadableStream` / `AsyncGenerator` |

**⚠️ Version mismatch warning:** Python SDK is on protocol v1.0; JS SDK is still on v0.3. A Python server and JS client may need the compat layer. Watch this gap.

## 2. Core Concept: The Agent Card

Every A2A agent publishes a JSON manifest at:
```
GET /.well-known/agent-card.json
```

This is the **only thing you need to discover an agent**. It tells you:
- Who the agent is (name, description, version, provider)
- What it can do (skills, with input/output MIME types and examples)
- How to reach it (endpoint URL, protocol binding)
- What auth it requires (security schemes)
- What features it supports (streaming, push notifications)

### Python Agent Card
```python
from a2a.types import AgentCard, AgentCapabilities, AgentInterface, AgentSkill

skill = AgentSkill(
    id='job-search',
    name='Job Search',
    description='Finds and scores job listings against a profile',
    tags=['jobs', 'career'],
    input_modes=['text/plain'],
    output_modes=['application/json'],
    examples=['Find AI security jobs matching my profile'],
)

agent_card = AgentCard(
    name='Hermes',
    description='Kirk\'s personal AI assistant',
    version='1.0.0',
    default_input_modes=['text/plain'],
    default_output_modes=['text/plain', 'application/json'],
    capabilities=AgentCapabilities(streaming=True, push_notifications=True, extended_agent_card=True),
    supported_interfaces=[
        AgentInterface(protocol_binding='JSONRPC', url='https://api.agentlocker.io/agents/hermes/')
    ],
    skills=[skill],
    security_schemes={'bearerAuth': {'type': 'http', 'scheme': 'bearer', 'bearerFormat': 'JWT'}},
    security=[{'bearerAuth': []}],
)
```

## 3. Data Model

### Message — one communication turn
```
Message {
  kind: "message"
  messageId: string (UUID)
  role: "user" | "agent"
  parts: Part[]          ← actual content
  contextId?: string     ← groups turns across a conversation
  taskId?: string        ← omit on first message, include on follow-ups
}
```

### Part — smallest content unit
```
TextPart  { kind: "text",  text: string }
FilePart  { kind: "file",  file: { bytes: base64 } | { uri: string }, mimeType?, name? }
DataPart  { kind: "data",  data: Record<string, unknown> }
```

### Task — stateful work unit
```
Task {
  kind: "task"
  id: string             ← server-generated UUID
  contextId: string      ← groups related tasks
  status: TaskStatus {
    state: "submitted" | "working" | "input-required" | "completed" | "canceled" | "failed"
    message?: Message    ← agent's status note
    timestamp?: string
  }
  artifacts?: Artifact[] ← outputs produced
  history?: Message[]
}
```

### Artifact — output produced by a task
```
Artifact {
  artifactId: string
  parts: Part[]
  name?: string
  description?: string
}
```

### Streaming events
```
TaskStatusUpdateEvent  { kind: "status-update",  taskId, contextId, status, final: boolean }
TaskArtifactUpdateEvent{ kind: "artifact-update", taskId, contextId, artifact, append?, lastChunk? }
```

## 4. Building a Server

### Python (Starlette)

```python
# 1. Implement AgentExecutor
from a2a.server.agent_execution import AgentExecutor, RequestContext
from a2a.server.events import EventQueue
from a2a.server.tasks import TaskUpdater
from a2a.helpers import get_message_text, new_task_from_user_message, new_text_message, new_text_part
from a2a.types.a2a_pb2 import TaskState

class HermesExecutor(AgentExecutor):
    async def execute(self, context: RequestContext, event_queue: EventQueue) -> None:
        # Get or create task
        task = context.current_task or new_task_from_user_message(context.message)
        if not context.current_task:
            await event_queue.enqueue_event(task)

        updater = TaskUpdater(event_queue, task.id, task.context_id)
        await updater.update_status(TaskState.TASK_STATE_WORKING, new_text_message('On it...'))

        # Your agent logic here
        query = get_message_text(context.message)
        result = await run_hermes(query)

        await updater.add_artifact(parts=[new_text_part(text=result, media_type='text/plain')])
        await updater.complete()  # → TASK_STATE_COMPLETED

    async def cancel(self, context: RequestContext, event_queue: EventQueue) -> None:
        raise NotImplementedError

# 2. Wire up the server
import uvicorn
from a2a.server.apps import A2AStarletteApplication
from a2a.server.request_handlers import DefaultRequestHandler
from a2a.server.tasks import InMemoryTaskStore

handler = DefaultRequestHandler(
    agent_executor=HermesExecutor(),
    task_store=InMemoryTaskStore(),
    agent_card=agent_card,          # Required in v1.0
)
server = A2AStarletteApplication(agent_card=agent_card, http_handler=handler)
uvicorn.run(server.build(), host='0.0.0.0', port=9999)
```

### TypeScript (Express)

```typescript
import express from 'express';
import { DefaultRequestHandler, InMemoryTaskStore, AgentExecutor, RequestContext, ExecutionEventBus } from '@a2a-js/sdk/server';
import { agentCardHandler, jsonRpcHandler, UserBuilder } from '@a2a-js/sdk/server/express';
import { AGENT_CARD_PATH } from '@a2a-js/sdk';

class HermesExecutor implements AgentExecutor {
  async execute(ctx: RequestContext, bus: ExecutionEventBus): Promise<void> {
    const { taskId, contextId, userMessage } = ctx;
    const input = userMessage.parts.filter(p => p.kind === 'text').map(p => (p as any).text).join(' ');
    const result = await runHermes(input);

    bus.publish({ kind: 'status-update', taskId, contextId, status: { state: 'working', timestamp: new Date().toISOString() }, final: false });
    bus.publish({ kind: 'artifact-update', taskId, contextId, artifact: { artifactId: 'result', parts: [{ kind: 'text', text: result }] }, lastChunk: true });
    bus.publish({ kind: 'status-update', taskId, contextId, status: { state: 'completed', timestamp: new Date().toISOString() }, final: true });
    bus.finished();
  }
  cancelTask = async () => {};
}

const handler = new DefaultRequestHandler(agentCard, new InMemoryTaskStore(), new HermesExecutor());
const app = express();
app.use(express.json());
app.use(`/${AGENT_CARD_PATH}`, agentCardHandler({ agentCardProvider: handler }));
app.use('/a2a/jsonrpc', jsonRpcHandler({ requestHandler: handler, userBuilder: UserBuilder.noAuthentication }));
app.listen(9999);
```

## 5. Building a Client

### Python
```python
import httpx
from a2a.client import A2ACardResolver, ClientConfig, create_client
from a2a.helpers import new_text_message
from a2a.types.a2a_pb2 import Role, SendMessageRequest

async with httpx.AsyncClient() as http:
    # 1. Discover agent
    card = await A2ACardResolver(httpx_client=http, base_url='http://localhost:9999').get_agent_card()

    # 2. Create client
    client = await create_client(agent=card, client_config=ClientConfig(streaming=True))

    # 3. Send message (streaming)
    request = SendMessageRequest(message=new_text_message('Find me AI security jobs', role=Role.ROLE_USER))
    async for chunk in client.send_message(request):
        print(chunk)

    await client.close()
```

import { v4 as uuidv4 } from 'uuid';

const factory = new ClientFactory();

// Non-streaming

for await (const event of stream) {
  if (event.kind === 'artifact-update') process.stdout.write((event.artifact.parts[0] as any).text);

## 6. Streaming — How It Works

Streaming uses **Server-Sent Events (SSE)** over HTTP. The client calls `message/stream` (JSON-RPC) instead of `message/send`. The same `execute()` method handles both — the framework decides which path based on the request.

**Event sequence for a typical task:**
```
Task (created)
  → TaskStatusUpdateEvent (state: working)
  → TaskArtifactUpdateEvent (chunk 1, append: true)
  → TaskArtifactUpdateEvent (chunk 2, append: true)
  → TaskArtifactUpdateEvent (final chunk, lastChunk: true)
  → TaskStatusUpdateEvent (state: completed, final: true)  ← closes stream
```

**Key rule:** `final: true` on a `TaskStatusUpdateEvent` closes the SSE stream. Always emit this or the client hangs.

## 7. Authentication

Auth is **declared** in the Agent Card (OpenAPI 3.0 security scheme style) but **not auto-enforced** by the SDK. You add your own middleware.

**Declaring in Agent Card:**
```python
security_schemes = {'bearerAuth': {'type': 'http', 'scheme': 'bearer', 'bearerFormat': 'JWT'}}
security = [{'bearerAuth': []}]  # OR list — at least one scheme must be satisfied
```

**Enforcing in Python (Starlette middleware):**
```python
from starlette.middleware.base import BaseHTTPMiddleware

class BearerAuthMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        if request.url.path == '/.well-known/agent-card.json':
            return await call_next(request)   # Card is always public
        auth = request.headers.get('Authorization', '')
        if not auth.startswith('Bearer ') or not validate_jwt(auth[7:]):
            return JSONResponse({'error': 'Unauthorized'}, 401)
        return await call_next(request)
```

**Enforcing in TypeScript (UserBuilder):**
```typescript
const authBuilder = UserBuilder.fromBearerToken(async (token) => {
  const user = await validateJWT(token);
  if (!user) throw new Error('Unauthorized');
  return { id: user.id };
});
app.use('/a2a/jsonrpc', jsonRpcHandler({ requestHandler: handler, userBuilder: authBuilder }));
```

**Extended Agent Card (auth-gated):** You can serve a richer card to authenticated clients (more skills, private endpoints). Set `capabilities.extended_agent_card: true` and pass `extended_agent_card` to `DefaultRequestHandler`. The public card stays minimal.

## 8. Next.js / Vercel Integration

This is the most relevant section for AgentLocker.

### Pattern A: Next.js as A2A Proxy (Dashboard → VPS Agent)
The Vercel dashboard doesn't run A2A itself — it proxies calls to the VPS agent. This is the right architecture for AgentLocker.

```typescript
// app/api/a2a/stream/route.ts
export const runtime = 'nodejs';   // ← required, not 'edge'
export const dynamic = 'force-dynamic';

export async function POST(req: NextRequest) {
  const { agentUrl, message, contextId } = await req.json();
  const factory = new ClientFactory();
  const client = await factory.createFromUrl(agentUrl);  // reads agent card

  const stream = new ReadableStream({
    async start(controller) {
      const enc = new TextEncoder();
      const agentStream = client.sendMessageStream({
        message: { kind: 'message', messageId: uuidv4(), role: 'user',
          parts: [{ kind: 'text', text: message }], ...(contextId && { contextId }) }
      });
      for await (const event of agentStream) {
        controller.enqueue(enc.encode(`data: ${JSON.stringify(event)}\n\n`));
      }
      controller.enqueue(enc.encode('data: [DONE]\n\n'));
      controller.close();
    }
  });

  return new Response(stream, {
    headers: {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache, no-transform',
      'X-Accel-Buffering': 'no',    // ← prevents Vercel proxy buffering
    }
  });
}
```

## 10. Gotchas & Pitfalls

### 🔴 Python v0.3 → v1.0 is a breaking change — types went Pydantic → Protobuf
```python
# v0.3 (old)
task.status.state == TaskState.completed           # lowercase enum
part = Part(root=TextPart(text="hello"))           # nested Pydantic

# v1.0 (current)
task.status.state == TaskState.TASK_STATE_COMPLETED  # SCREAMING_SNAKE_CASE
part = Part(text="hello")                            # flat Protobuf
```

### 🔴 Python v1.0: `AgentCard.url` is gone — moved to `AgentInterface`
```python
# Wrong in v1.0:
AgentCard(url='http://localhost:9999/', ...)
# Correct:
AgentCard(supported_interfaces=[AgentInterface(protocol_binding='JSONRPC', url='http://localhost:9999/')], ...)
```

### 🔴 `agent_card` is now required in `DefaultRequestHandler` (Python v1.0)
```python
DefaultRequestHandler(agent_executor=..., task_store=..., agent_card=agent_card)  # agent_card required
```

### 🔴 JS SDK is still on protocol v0.3 — Python is on v1.0
If you build a Python server and call it from a JS client, test the compat layer (`a2a.compat.v0_3`).

### 🟡 Auth is declared but NOT enforced by the SDK
You must add your own middleware. The SDK will happily serve requests without tokens unless you block them.

### 🟡 Task IDOR — any client can access any task by ID
`InMemoryTaskStore` and `DatabaseTaskStore` look up by ID only. Configure `owner_resolver` in v1.0 or add your own scoping middleware.

### 🟡 Push notification SSRF risk
Webhook URLs are passed to `httpx.post()` without URL validation. Validate webhook URLs server-side before accepting them.

### 🟡 `InMemoryPushNotificationConfigStore` has a multi-owner bug
`get_all_for_task()` falls back to empty owner context in multi-owner deployments, silently dropping push notifications. Override this method in multi-owner setups.

### 🟡 Streaming: always emit `final: true` + `finished()`
Python: `await updater.complete()` or `update_status(..., final=True)`  
TypeScript: `final: true` on the last status event + `bus.finished()`  
Missing either → client hangs indefinitely.

### 🟡 Prompt injection via Agent Card data
Agent Card fields (`name`, `description`, `skills.description`) coming from remote agents are untrusted. Sanitize before passing into LLM prompts.

### 🟢 The entire Python SDK is async-only
Every method is `async def`. Entry point must use `asyncio.run()`.

### 🟢 `examples` field moved — now lives on `AgentSkill`, not `AgentCard`

## 11. AgentLocker-Specific Notes

### What this means for the hub vision
- The Vercel dashboard is an **A2A client** — it discovers agents via their cards and sends tasks
- Hermes on VPS becomes an **A2A server** — expose `AgentExecutor` wrapping existing gateway capabilities
- RSR becomes an **A2A server** — same pattern, different VPS
- AgentLocker maintains a **registry** of agent card URLs — users add new agents by URL, dashboard discovers the card

### Minimal Hermes A2A exposure path
1. Add `A2AStarletteApplication` to Hermes gateway (or run as a sidecar on a new port)
2. Wrap existing Hermes tools as `AgentSkill` entries in the card
3. Implement `AgentExecutor.execute()` that forwards to existing Hermes handler
4. nginx proxies `api.agentlocker.io/agents/hermes/` → Hermes A2A port

### InMemoryTaskStore is fine for VPS (unlike Vercel)
On the VPS, Hermes runs as a persistent process — InMemoryTaskStore survives across requests. No Redis needed on the agent side. Vercel (the dashboard proxy) is where you'd need Redis if tracking task state client-side.

## 13. Deep-Dive Analysis of AgentiCat (Next.js 16 Dashboard Reference)

While `julianduque/agenticat` is a brand-new repository with **zero stars** (rightfully triggering caution before running or cloning locally), an inspection of its codebase reveals a highly robust, enterprise-grade Next.js App Router implementation of the A2A SDK. 

Instead of running untrusted code, we can **steal its exact architectural design patterns** for our AgentLocker Vercel dashboard.

### Key Architectural Takeaways

#### A. Secure Next.js Streaming Proxy Router (`app/api/agents/stream/route.ts`)
The biggest challenge of a serverless dashboard is securely proxying long-running Agent SSE (Server-Sent Events) streams without exposing API credentials to the client browser or getting choked by Vercel's edge proxy buffering.

AgentiCat handles this beautifully with a dual-mode streaming proxy:
1. **SDK-driven Mode:** When a `cardUrl` is provided, it uses the official client factory to fetch the card, build an authenticated channel, and stream messages:
   ```typescript
   const factory = new ClientFactory();
   const client = await factory.createFromUrl(cardUrl, authToken);
   const stream = new ReadableStream({
     async start(controller) {
       for await (const event of client.sendMessageStream(params)) {
         controller.enqueue(new TextEncoder().encode(`data: ${JSON.stringify(event)}\n\n`));
       }
       controller.enqueue(new TextEncoder().encode("data: [DONE]\n\n"));
       controller.close();
     }
   });
   ```
2. **Direct Pass-Through Mode:** When hitting raw endpoints or custom HTTP/header-based authentication that the official SDK client chokes on, it bypasses the SDK entirely. It makes a direct Node `fetch` call with an `AbortSignal` timeout and pipes the raw upstream response stream directly back to the client:
   ```typescript
   const upstreamResponse = await directStreamingCall(endpointUrl, method, params, authHeaders);
   return new Response(upstreamResponse.body, {
     headers: {
       "Content-Type": "text/event-stream",
       "Cache-Control": "no-cache",
       "Connection": "keep-alive",
     },
   });
   ```

#### B. Defeating Edge Proxy Buffering
By default, Vercel or cloud proxies can buffer SSE chunks, defeating the real-time "typing" experience. AgentiCat forces progressive streaming by adding downstream headers:
*   `Cache-Control: no-cache, no-transform`
*   `X-Accel-Buffering: no` (specifically instructs Nginx/proxies not to buffer the stream)

#### C. Handling Serverless State Limitations
*   **The Ephemeral Task Store Trap:** The SDK's default `InMemoryTaskStore` works fine for persistent setups (like our VPS), but fails on serverless edge functions because state is wiped between API invocations.
*   **The Solution:** For our AgentLocker dashboard, client-side conversation history or a serverless Redis store (such as Upstash) must be utilized to maintain task states if navigating away from the chat screen.
